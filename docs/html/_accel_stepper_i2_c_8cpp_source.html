52 bool AccelStepperI2C::readResult(uint8_t numBytes) {
53  buf.reset();
54  if (Wire.requestFrom(address, numBytes) > 0) { // success
55  while (buf.idx <= numBytes and buf.idx < buf.maxLen) {
56  buf.write(Wire.read());
57  //buf.buffer[buf.idx++] = Wire.read();
58  }
59  resultOK = buf.checkCRC8();
60  } else { // some transmission error
61  resultOK = false;
62  }
63  buf.reset(); // reset for reading
64  return resultOK;
65 }
