103  /* The CNC shield by default pulls the four drivers' _EN_ pins high, making them inactive.
104  You can enable theim either by installing a jumper next to the reset button between EN and GND
105  *or* by software. As all four EN are connected to Arduino pin 8, we have to tell Accelstepper
106  about that pin and to call enableOutputs(). Before that, we have to tell it that _EN_ is active low
107  with setPinsInverted().
108  */
109  bool res = true;
110 
111  X->setEnablePin(8); res &= X->sentOK;
112 // directionInvert, stepInvert, enableInvert
113  X->setPinsInverted(false, false, true); res &= X->sentOK;
114  X->enableOutputs(); res &= X->sentOK;
115 
116  Y->setEnablePin(8); res &= Y->sentOK;
117  Y->setPinsInverted(false, false, true); res &= Y->sentOK;
118  Y->enableOutputs(); res &= Y->sentOK;
119 
120  Z->setEnablePin(8); res &= Z->sentOK;
121  Z->setPinsInverted(false, false, true); res &= Z->sentOK;
122  Z->enableOutputs(); res &= Z->sentOK;
123 
124  if (!res) {
125  Serial.print("transmission error");
127  }
128 
129  // tailor MaxSpeeds and acceleration to your steppers, here X is a Nema-14, Y and Z are geared 28BYJ-48's
130  X->setMaxSpeed(2000); Y->setMaxSpeed(500); Z->setMaxSpeed(500);
131  X->setAcceleration(200); Y->setAcceleration(300); Z->setAcceleration(300);
132 
133  // install endstop switch
134  X->setEndstopPin(endstopXpin, true, true); // activeLow works nicely together with internal pullup
135  X->enableEndstops(); // true by default
136 
137  X->moveTo(5000); Y->moveTo(8000); Z->moveTo(12000); // set targets
138  X->runState(); Y->runState(); Z->runState(); // start the three state machines

152 void loop() {
153  Serial.print("X endstops="); Serial.print(X->endstops()); Serial.print(" - "); printStats(X); //Serial.print("Y - "); printStats(Y); Serial.print("Z - "); printStats(Z);
154  delay (1000);
155  if (abs(X->distanceToGo()) == 0) {
156  X->moveTo(0 - X->currentPosition());
157  // after the target has been reached, the state machine resets itself to state_stopped. So we must start it again.
158  X->runState();
159  }
160 
161  if (abs(Y->distanceToGo()) == 0) {
162  Y->moveTo(0 - Y->currentPosition()); Y->runState();
163  }
164 
165  if (abs(Z->distanceToGo()) == 0) {
166  Z->moveTo(0 - Z->currentPosition()); Z->runState();
167  }
