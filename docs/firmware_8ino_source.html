81 // Read a stored I2C address from EEPROM. If there is none, use default.
82 uint8_t retrieveI2C_address() {
83  SimpleBuffer b;
84  b.init(8);
85  // read 6 bytes from eeprom
86  for (byte i = 0; i < 6; i++) {
87  b.buffer[i] = EEPROM.read(EEPROM_OFFSET_I2C_ADDRESS + i);
88  }
89  b.idx = 6; // [0]=CRC8; [1-4]=marker; [5]=address
90  uint32_t markerTest; b.read(markerTest);
91  if (b.checkCRC8() and (markerTest == eepromI2CaddressMarker)) {
92  uint8_t storedAddress; b.read(storedAddress);
93  return storedAddress;
94  } else {
95  return I2C_defaultAddress;
96  }
97 }

220 /*
221  Read endstop(s) for stepper s.
222  Returns bit pattern, last added endstop is LSB; 0 for inactive, 1 for active (taking activeLow in account)
223  returns 0x0, if no endstops set
224 */
225 uint8_t readEndstops(uint8_t s) {
226  uint8_t res = 0;
227  for (uint8_t i = 0; i < steppers[s].numEndstops; i++) {
228  //log(i); log(" p");
229  //log(digitalRead(steppers[s].endstops[i].pin)); log(" ");
230  res = (res << 1) | (digitalRead(steppers[s].endstops[i].pin) ^ steppers[s].endstops[i].activeLow); // xor
231  }
232  //log(" res="); log(res); log(" ");
233  return res;
234 }

298  if (steppers[i].stepper->distanceToGo() == 0)

303  }
304  break;

308  } // switch
309 
310  if (timeToCheckTheEndstops and steppers[i].endstopsEnabled) {
311  if (readEndstops(i) != 0) {

313  triggerInterrupt(i);
314  }
315  }
316 
317  } // for

320  // ### Move into for-loop, so that it's checked more often?

324  }

332  }

336 #endif
337 
338 #ifdef DEBUG_AccelStepperI2C

342  }
343 #endif
344 
345  cycles++;
